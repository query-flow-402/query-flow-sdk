Quick Example

import { QueryFlowClient } from "@queryflow-402/sdk";

const client = new QueryFlowClient(process.env.PRIVATE_KEY);

const result = await client.market({
  assets: ["BTC", "ETH"],
  timeframe: "24h",
});

console.log(`Sentiment: ${result.sentiment.score}/100`);
